134     /**
135      * Add Insulin to database and list
136      */
137     private void addInsulin(){
138         Insulin insulin = new Insulin(add_insulin_text_field.getText());
139         insulinDatabase.add(insulin);
140         
141         addInsulinToList(insulin, true);
142     }
143     
144     /**
145      * Add Insulin to list
146      * @param insulin
147      * @param refresh 
148      */
149     private void addInsulinToList(final Insulin insulin, boolean refresh){
150         GridBagConstraints gbc = new GridBagConstraints();
151         gbc.gridwidth = GridBagConstraints.REMAINDER;
152         gbc.weightx = 1;
153         gbc.fill = GridBagConstraints.HORIZONTAL;
154         
155         SimpleDateFormat formatter = new SimpleDateFormat("hh:mm", Locale.ENGLISH);
156         JPanel paneeli = new JPanel();
157         paneeli.add(new JLabel(insulin.getName() + " " + insulin.getPerCarbohydrate()));
158         
159         // Remove -Button
160         final JButton btnRemove = new JButton("Poista");
161         btnRemove.addActionListener(new ActionListener()
162         {
163             @Override
164             public void actionPerformed(ActionEvent e)
165             {
166                 insulinDatabase.delete(insulin);
167                 insulins_list.remove(btnRemove.getParent());
168                 insulins_list.revalidate();
169                 insulins_list.repaint();
170             }
171         });
172         paneeli.add(btnRemove);
173         
174         paneeli.setBorder(new MatteBorder(0, 0, 1, 0, Color.GRAY));
175         this.insulins_list.add(paneeli, gbc, 0);
176         
177         if(refresh){
178             this.insulins_list.revalidate();
179             this.insulins_list.repaint();
180         }
181     }
182     
183     /**
184      * Update all Insulins to list
185      */
186     private void updateInsulinsList(){
187         ArrayList<Insulin> il = this.insulinDatabase.all();
188         
189         this.insulins_list.removeAll();
190         
191         GridBagConstraints gbcTemp = new GridBagConstraints();
192         gbcTemp.gridwidth = GridBagConstraints.REMAINDER;
193         gbcTemp.weightx = 1;
194         gbcTemp.weighty = 1;
195         this.insulins_list.add(new JPanel(), gbcTemp);
196         
197         for (final Insulin i : il) {
198             addInsulinToList(i, false);
199         }
200         
201         this.insulins_list.revalidate();
202         this.insulins_list.repaint();
203     }
