267     private void addBloodSugar(){
268         BloodSugar bloodSugar = new BloodSugar(this.bloodsugar.getText());
269         bloodSugarDatabase.add(bloodSugar);
270         
271         updateRecentBloodSugars();
272         this.bloodsugar.setText("");
273     }

275     private void updateRecentBloodSugars(){
276         ArrayList<BloodSugar> bs = bloodSugarDatabase.all();
277         
278         GridBagConstraints gbc = new GridBagConstraints();
279         gbc.gridwidth = GridBagConstraints.REMAINDER;
280         gbc.weightx = 1;
281         gbc.fill = GridBagConstraints.HORIZONTAL;
282         
283         for (final BloodSugar b : bs) {
284             SimpleDateFormat formatter = new SimpleDateFormat("hh:mm", Locale.ENGLISH);
285             JPanel paneeli = new JPanel();
286             paneeli.add(new JLabel(formatter.format(b.getMeasuredAt()) + " - " + b.getBloodSugar()));
287             final JButton btn = new JButton(b.getInsulinName() + " - " + b.getInsulinAmount());
288             if(b.isInsulinTaken()){
289                 btn.setEnabled(false);
290             }
291             btn.addActionListener(new ActionListener()
292             {
293                 @Override
294                 public void actionPerformed(ActionEvent e)
295                 {
296                     b.takeInsulin();
297                     btn.setEnabled(false);
298                     bloodSugarDatabase.save();
299                 }
300             });
301             paneeli.add(btn);
302             paneeli.setBorder(new MatteBorder(0, 0, 1, 0, Color.GRAY));
303             this.mainlist.add(paneeli, gbc, 0);
304         }
305         
306         this.mainlist.validate();
307         this.mainlist.repaint();
308     }
