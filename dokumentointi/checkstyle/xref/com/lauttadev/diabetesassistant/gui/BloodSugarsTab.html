75      public void updateInsulinsComboBox(){
76          ArrayList<Insulin> insulins = this.insulinDatabase.all();
77          ArrayList<String> list = new ArrayList();
78          list.add("Ei insuliiniä");
79          
80          for(int i = 0; i < insulins.size(); i++){
81              list.add(insulins.get(i).toString());
82          }
83          
84          this.insulins_combo_box.setModel(new javax.swing.DefaultComboBoxModel(list.toArray()));
85          this.insulins_combo_box.setSelectedIndex(0);
86          
87          this.insulins_combo_box.revalidate();
88          this.insulins_combo_box.repaint();
89      }

250     private boolean checkIfValidBloodSugar(){
251         return !this.bloodsugar.getText().equals("") && ("HI".equals(bloodsugar.getText().toUpperCase()) || "LO".equals(bloodsugar.getText().toUpperCase()) || bloodsugar.getText().matches("^[0-9]+\\.?([0-9]+)?$"));
252     }

268     /**
269      * Add BloodSugar to database and to the list
270      */
271     private void addBloodSugar(){
272         if(this.checkIfValidBloodSugar()){
273             BloodSugar bloodSugar = new BloodSugar(this.bloodsugar.getText());
274 
275             bloodSugarDatabase.add(bloodSugar);
276             this.bloodsugar.setText("");
277             addBloodSugarToList(bloodSugar, true);
278             bloodSugarDatabase.save();
279         } else {
280             this.add_bloodsugar.setEnabled(false);
281         }
282     }

284     /**
285      * Add given BloodSugar into the list of recent BloodSugars
286      * @param bloodsugar 
287      */
288     private void addBloodSugarToList(final BloodSugar bloodsugar, boolean refresh){
289         GridBagConstraints gbc = new GridBagConstraints();
290         gbc.gridwidth = GridBagConstraints.REMAINDER;
291         gbc.weightx = 1;
292         gbc.fill = GridBagConstraints.HORIZONTAL;
293         
294         SimpleDateFormat formatter = new SimpleDateFormat("hh:mm", Locale.ENGLISH);
295         JPanel paneeli = new JPanel();
296         paneeli.add(new JLabel(formatter.format(bloodsugar.getMeasuredAt()) + " - " + bloodsugar.getBloodSugar()));
297         
298         if(bloodsugar.getInsulinName() != null){
299             // Take Insulin -button
300             final JButton btn = new JButton(bloodsugar.getInsulinName() + " - " + bloodsugar.getInsulinAmount());
301             paneeli.add(btn);
302             btn.addActionListener(new ActionListener()
303             {
304                 @Override
305                 public void actionPerformed(ActionEvent e)
306                 {
307                     bloodsugar.takeInsulin();
308                     btn.setEnabled(false);
309                     bloodSugarDatabase.save();
310                 }
311             });
312             if(bloodsugar.isInsulinTaken()){
313                 btn.setEnabled(false);
314             }
315         }
316 
317         // Remove -button
318         final JButton btnRemove = new JButton("Poista");
319         paneeli.add(btnRemove);
320         btnRemove.addActionListener(new ActionListener()
321         {
322             @Override
323             public void actionPerformed(ActionEvent e)
324             {
325                 bloodSugarDatabase.delete(bloodsugar);
326                 bloodsugars_list.remove(btnRemove.getParent());
327                 bloodsugars_list.revalidate();
328                 bloodsugars_list.repaint();
329             }
330         });
331         
332         paneeli.setBorder(new MatteBorder(0, 0, 1, 0, Color.GRAY));
333         this.bloodsugars_list.add(paneeli, gbc, 0);
334         
335         // Refresh only if boolean true
336         if(refresh){
337             bloodsugars_list.revalidate();
338             bloodsugars_list.repaint();
339         }
340     }

342     /**
343      * Update list of recent BloodSugars
344      */
345     private void updateRecentBloodSugars(){
346         ArrayList<BloodSugar> bsl = bloodSugarDatabase.all();
347         
348         this.bloodsugars_list.removeAll();
349         
350         GridBagConstraints gbcTemp = new GridBagConstraints();
351         gbcTemp.gridwidth = GridBagConstraints.REMAINDER;
352         gbcTemp.weightx = 1;
353         gbcTemp.weighty = 1;
354         this.bloodsugars_list.add(new JPanel(), gbcTemp);
355 
356         for (final BloodSugar bs : bsl) {
357             this.addBloodSugarToList(bs, false);
358         }
359         
360         this.bloodsugars_list.revalidate();
361         this.bloodsugars_list.repaint();
362     }
