237     /**
238      * Add given BloodSugar into the list of recent BloodSugars
239      * @param bloodsugar 
240      */
241     private void addBloodSugarToList(final BloodSugar bloodsugar, boolean refresh){
242         GridBagConstraints gbc = new GridBagConstraints();
243         gbc.gridwidth = GridBagConstraints.REMAINDER;
244         gbc.weightx = 1;
245         gbc.fill = GridBagConstraints.HORIZONTAL;
246         
247         SimpleDateFormat formatter = new SimpleDateFormat("hh:mm", Locale.ENGLISH);
248         JPanel paneeli = new JPanel();
249         paneeli.add(new JLabel(formatter.format(bloodsugar.getMeasuredAt()) + " - " + bloodsugar.getBloodSugar()));
250         
251         if(bloodsugar.getInsulinName() != null){
252             // Take Insulin -button
253             final JButton btn = new JButton(bloodsugar.getInsulinName() + " - " + bloodsugar.getInsulinAmount());
254             paneeli.add(btn);
255             btn.addActionListener(new ActionListener()
256             {
257                 @Override
258                 public void actionPerformed(ActionEvent e)
259                 {
260                     bloodsugar.takeInsulin();
261                     btn.setEnabled(false);
262                     bloodSugarDatabase.save();
263                 }
264             });
265             if(bloodsugar.isInsulinTaken()){
266                 btn.setEnabled(false);
267             }
268         }
269 
270         // Remove -button
271         final JButton btnRemove = new JButton("Poista");
272         paneeli.add(btnRemove);
273         btnRemove.addActionListener(new ActionListener()
274         {
275             @Override
276             public void actionPerformed(ActionEvent e)
277             {
278                 bloodSugarDatabase.delete(bloodsugar);
279                 mainlist.remove(btnRemove.getParent());
280                 mainlist.revalidate();
281                 mainlist.repaint();
282             }
283         });
284         
285         paneeli.setBorder(new MatteBorder(0, 0, 1, 0, Color.GRAY));
286         this.mainlist.add(paneeli, gbc, 0);
287         
288         // Refresh only if boolean true
289         if(refresh){
290             mainlist.revalidate();
291             mainlist.repaint();
292         }
293     }

295     /**
296      * Update list of recent BloodSugars
297      */
298     private void updateRecentBloodSugars(){
299         ArrayList<BloodSugar> bsl = bloodSugarDatabase.all();
300         
301         this.mainlist.removeAll();
302         
303         GridBagConstraints gbcTemp = new GridBagConstraints();
304         gbcTemp.gridwidth = GridBagConstraints.REMAINDER;
305         gbcTemp.weightx = 1;
306         gbcTemp.weighty = 1;
307         this.mainlist.add(new JPanel(), gbcTemp);
308 
309         for (final BloodSugar bs : bsl) {
310             this.addBloodSugarToList(bs, false);
311         }
312         
313         this.mainlist.revalidate();
314         this.mainlist.repaint();
315     }
